<script>
    // 1. Funcție pentru selectarea automată a telefonului din card în formular
    function selecteazaTelefon(numeTelefon) {
        let select = document.getElementById('telefon-selectat');
        if(numeTelefon === 'ATech Pro Max') select.value = "4500";
        if(numeTelefon === 'ATech Air') select.value = "3200";
        if(numeTelefon === 'ATech Lite') select.value = "1800";
        
        document.getElementById('comanda').scrollIntoView({behavior: 'smooth'});
    }

    // 2. SARCINA PRINCIPALĂ: Procesarea informației din formular
    function proceseazaComanda() {
        // Preluăm datele introduse de utilizator
        let nume = document.getElementById('nume').value;
        let pretUnitat = parseFloat(document.getElementById('telefon-selectat').value);
        let cantitate = parseInt(document.getElementById('cantitate').value);
        let codPromo = document.getElementById('promo').value.trim();
        
        // Validare simplă: verificăm dacă utilizatorul a introdus numele
        if(nume === "") {
            alert("Vă rugăm să introduceți numele pentru a procesa comanda!");
            return;
        }

        // Calculăm prețul de bază
        let pretTotal = pretUnitat * cantitate;
        let mesajReducere = "";

        // Verificăm dacă codul de reducere este corect (ATECH20 oferă 20% reducere)
        if(codPromo === "ATECH20") {
            pretTotal = pretTotal * 0.8; // Se scade 20%
            mesajReducere = " (A fost aplicată o reducere de 20%!)";
        }

        // Formăm răspunsul și îl afișăm în pagină
        let cutieRezultat = document.getElementById('rezultat-comanda');
        cutieRezultat.innerHTML = "Salut, " + nume + "! Comanda ta a fost procesată. Preț final: " + pretTotal + " RON" + mesajReducere + ".";
        cutieRezultat.style.display = "block"; // Facem textul vizibil
    }

    // 3. SARCINA SUPLIMENTARĂ: Efect vizual de schimbare a temei (Dark/Light Mode)
    function schimbaTema() {
        let corpPagina = document.body;
        corpPagina.classList.toggle("dark-mode"); // Adaugă sau șterge clasa dark-mode
        
        // Schimbăm textul de pe buton în funcție de mod
        let buton = document.querySelector('.dark-mode-btn');
        if(corpPagina.classList.contains("dark-mode")) {
            buton.innerHTML = "Schimbă Tema ☀️";
        } else {
            buton.innerHTML = "Schimbă Tema 🌙";
        }
    }
</script>
